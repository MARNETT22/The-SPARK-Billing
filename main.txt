def load_settings():
    default = {
        "logo_width": 30,
        "watermark_width": 120,
        "watermark_alpha": 0.15,
        "signature_width": 24,
        "signature_box_width": 50,
        "signature_offset_x": 0,
        "signature_offset_y": 18
    }
    try:
        if os.path.exists(SETTINGS_FILE):
            with open(SETTINGS_FILE, "r", encoding="utf-8") as f:
                data = json.load(f)
                default.update(data or {})
    except (json.JSONDecodeError, IOError) as e:
        print(f"Warning: Failed to load settings.json: {e}")
    except Exception as e:
        print(f"Unexpected error loading settings: {e}")
    return default
def save_settings(data):
    try:
        with open(SETTINGS_FILE, "w", encoding="utf-8") as f:
            json.dump(data, f, ensure_ascii=False, indent=2)
        return True
    except Exception as e:
        print(f"Error saving settings: {e}")
        return False
def resolve_image(candidates):
    for n in candidates:
        p = os.path.join(BASE_DIR, n)
        if os.path.exists(p):
            return p
    return None
def add_image_safe(pdf, path, x, y, w):
    if not path:
        return
    try:
        pdf.image(path, x, y, w)
    except Exception:
        if PILImage:
            tmp = None
            try:
                fd, tmp = tempfile.mkstemp(suffix=".jpg")
                os.close(fd)
                img = PILImage.open(path).convert("RGB")
                img.save(tmp, "JPEG")
                pdf.image(tmp, x, y, w)
            finally:
                if tmp and os.path.exists(tmp):
                    try:
                        os.remove(tmp)
                    except:
                        pass
LOGO_FILE = resolve_image(["logo.png", "Logo.png", "logo.jpg", "logo.jpeg"])
SIGN_FILE = resolve_image(["DigitalSignature.png", "digital signature.png", "digitalsignature.png", "signature.png", "signature.jpg", "digitalsignature.jpg"])
def add_watermark_safe(pdf, path, w, alpha=0.15):
    if not path:
        return
    try:
        if PILImage:
            tmp = None
            try:
                fd, tmp = tempfile.mkstemp(suffix=".jpg")
                os.close(fd)
                img = PILImage.open(path).convert("RGB")
                white = PILImage.new("RGB", img.size, (255, 255, 255))
                faded = PILImage.blend(white, img, alpha if 0.0 <= alpha <= 1.0 else 0.15)
                faded.save(tmp, "JPEG")
                inner_w = pdf.w - pdf.l_margin - pdf.r_margin
                inner_h = pdf.h - pdf.t_margin - pdf.b_margin
                x = pdf.l_margin + (inner_w - w) / 2
                y = pdf.t_margin + (inner_h - w) / 2
                pdf.image(tmp, x, y, w)
            finally:
                if tmp and os.path.exists(tmp):
                    try:
                        os.remove(tmp)
                    except:
                        pass
        else:
            inner_w = pdf.w - pdf.l_margin - pdf.r_margin
            inner_h = pdf.h - pdf.t_margin - pdf.b_margin
            x = pdf.l_margin + (inner_w - w) / 2
            y = pdf.t_margin + (inner_h - w) / 2
            pdf.image(path, x, y, w)
    except Exception as e:
        print(f"Error adding watermark: {e}")

def generate_invoice_id():
    dt = datetime.now().strftime("%Y%m%d%H%M%S")
    try:
        conn = sqlite3.connect(DB_FILE)
        cursor = conn.cursor()
        cursor.execute("SELECT COUNT(*) FROM invoices")
        count = cursor.fetchone()[0]
        conn.close()
        seq = count + 1
    except:
        seq = 1
    return f"{dt}{seq}"

def calculate_verification_hash(inv_id):
    """Calculates a unique security hash for an invoice ID."""
    SECRET_SALT = "SPARK_SECURE_2026"
    secure_str = f"{inv_id}{SECRET_SALT}"
    return hashlib.sha256(secure_str.encode()).hexdigest()[:12].upper()

def add_security_elements(pdf, inv_id):
    """Adds security features to make the invoice harder to forge."""
    try:
        # 1. Background Security Text (Multiple faint rows)
        pdf.set_font("Times", 'B', 35)
        pdf.set_text_color(248, 248, 248) # Extremely faint gray
        for i in range(4):
            pdf.text(15, 80 + i*60, "OFFICIAL INVOICE")
        
        # 2. Unique Security Hash at the bottom
        security_hash = calculate_verification_hash(inv_id)
        
        pdf.set_font("Times", 'I', 7)
        pdf.set_text_color(180, 180, 180)
        pdf.text(10, pdf.h - 8, f"Verification ID: {inv_id}-{security_hash} | Authenticated by The SPARK Education Centre")
        
        # 3. Small "Original" mark
        pdf.set_font("Times", 'B', 8)
        pdf.set_text_color(200, 200, 200)
        pdf.text(pdf.w - 40, 15, "ORIGINAL DOCUMENT")
    except:
        pass

        """Handles application closing, including auto-backup."""
        self.auto_backup_on_exit()

    def auto_backup_on_exit(self):
        """Silently creates a backup of the SQLite database on exit."""
        if not os.path.exists(DB_FILE):
            return 
        try:
            backup_dir = os.path.join(BASE_DIR, "Backup Folder")
            os.makedirs(backup_dir, exist_ok=True)
            
            timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
            unique_id = hashlib.sha1(os.urandom(8)).hexdigest()[:6]
            backup_name = f"DB_Backup_{timestamp}_{unique_id}.db"
            dest = os.path.join(backup_dir, backup_name)
            
            shutil.copy2(DB_FILE, dest)
        except Exception as e:
            print(f"Auto-backup on exit failed: {e}")

    def init_db(self):
        try:
            conn = sqlite3.connect(DB_FILE)
            cursor = conn.cursor()
            cursor.execute("""
                CREATE TABLE IF NOT EXISTS invoices (
                    id TEXT PRIMARY KEY,
                    date TEXT,
                    name TEXT,
                    items_details TEXT,
                    month TEXT,
                    year TEXT,
                    total REAL,
                    advance REAL DEFAULT 0,
                    advance_item TEXT,
                    remarks TEXT
                )
            """)
            conn.commit()
            
            # Migration: add advance column if it doesn't exist
            try:
                cursor.execute("ALTER TABLE invoices ADD COLUMN advance REAL DEFAULT 0")
                conn.commit()
            except:
                pass
            
            try:
                cursor.execute("ALTER TABLE invoices ADD COLUMN advance_item TEXT")
                conn.commit()
            except:
                pass
            
            # Migration from CSV if exists and SQLite is empty
            cursor.execute("SELECT COUNT(*) FROM invoices")
            if cursor.fetchone()[0] == 0 and os.path.exists(CSV_MIGRATE_FILE):
                try:
                    with open(CSV_MIGRATE_FILE, mode='r', encoding='utf-8') as f:
                        reader = csv.DictReader(f)
                        for row in reader:
                            cursor.execute("""
                                INSERT INTO invoices (id, date, name, items_details, month, year, total, remarks)
                                VALUES (?, ?, ?, ?, ?, ?, ?, ?)
                            """, (row["ID"], row["Date"], row["Name"], row["Items_Details"], row["Month"], row["Year"], float(row["Total"]), row.get("Remarks", "")))
                    conn.commit()
                    # Mark CSV as migrated
                    os.rename(CSV_MIGRATE_FILE, CSV_MIGRATE_FILE + ".migrated")
                except Exception as ex:
                    print(f"Migration error: {ex}")
            
            conn.close()
        except Exception as e:
            messagebox.showerror("Database Error", f"Failed to initialize database: {e}")

    def add_item_to_list(self):
        cat, sub, amt, adv = self.ent_item_desc.get(), self.ent_sub_item.get(), self.ent_item_amt.get(), self.ent_item_advance.get()
        try:
            val = float(str(amt).replace(",", ""))
            adv_val = float(str(adv).replace(",", "")) if adv.strip() else 0.0
            display_full = f"{cat} ({sub})" if sub.strip() else cat
            stu = self.ent_item_student.get().strip()
            cat_display = cat if not stu else f"{cat} - {stu}"
            if self.editing_id:
                children = list(self.item_tree.get_children())
                if self.editing_id in children:
                    idx = children.index(self.editing_id)
                    self.current_items[idx] = {"cat": cat, "sub": sub, "student": stu, "desc": display_full, "amt": val, "adv": adv_val}
                    self.item_tree.item(self.editing_id, values=(cat_display, sub, f"{val:,.0f}", f"{adv_val:,.0f}"))
                self.editing_id = None
                self.btn_add_update.config(text="+ ADD TO LIST")
            else:
                self.current_items.append({"cat": cat, "sub": sub, "student": stu, "desc": display_full, "amt": val, "adv": adv_val})
                count = len(self.item_tree.get_children())
                tag = "odd" if count % 2 == 0 else "even"
                self.item_tree.tag_configure("odd", background="#fafafa")
                self.item_tree.tag_configure("even", background="#ffffff")
                self.item_tree.insert("", "end", values=(cat_display, sub, f"{val:,.0f}", f"{adv_val:,.0f}"), tags=(tag,))
            self.update_total_display()
            self.ent_item_amt.delete(0, tk.END); self.ent_item_amt.insert(0, "0")
            self.ent_item_advance.delete(0, tk.END); self.ent_item_advance.insert(0, "0")
            self.ent_item_student.delete(0, tk.END)
        except: messagebox.showerror("Error", "Check Amount")
        try:
            self.update_finalize_state()
        except:
            pass

    def format_amount_text(self, txt):
        try:
            raw = str(txt).replace(",", "").strip()
            if raw == "": return ""
            if "." in raw:
                int_part, dec_part = raw.split(".", 1)
                int_fmt = f"{int(int_part):,}"
                return f"{int_fmt}.{dec_part}"
            else:
                return f"{int(raw):,}"
        except:
            return txt

    def finalize_payment(self):
        name, month, year = self.ent_name.get().strip(), self.ent_month.get(), self.ent_year.get()
        if not name or not self.current_items: return
        
        subtotal = sum(i["amt"] for i in self.current_items)
        advance = sum(i.get("adv", 0.0) for i in self.current_items)
        total = subtotal - advance
        
        # Format: desc:amt:adv
        items_str = " | ".join([f"{i['desc']}:{i['amt']}:{i.get('adv', 0.0)}" for i in self.current_items])
        
        inv_id = generate_invoice_id()
        try:
            conn = sqlite3.connect(DB_FILE)
            cursor = conn.cursor()
            cursor.execute("""
                INSERT INTO invoices (id, date, name, items_details, month, year, total, advance, remarks)
                VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
            """, (inv_id, datetime.now().strftime("%Y-%m-%d"), name, items_str, month, year, total, advance, ""))
            conn.commit()
            conn.close()
            
            self.export_invoice_pdf(inv_id, name, month, year, subtotal, advance)
            messagebox.showinfo("Success", f"Invoice #{inv_id} created!")
            self.clear_all()
        except Exception as e:
            messagebox.showerror("Database Error", f"Failed to save invoice: {e}")

    def generate_monthly_report_pdf(self):
        year = str(self.ent_year.get())
        month_filter = self.rep_month.get()
        if not os.path.exists(DB_FILE):
            messagebox.showerror("Report Error", "No database found")
            return
        
        try:
            conn = sqlite3.connect(DB_FILE)
            cursor = conn.cursor()
            
            if month_filter == "All":
                cursor.execute("SELECT month, name, total FROM invoices WHERE year = ?", (year,))
            else:
                cursor.execute("SELECT month, name, total FROM invoices WHERE year = ? AND month = ?", (year, month_filter))
            
            rows = cursor.fetchall()
            conn.close()
            
            months_order = ["January","February","March","April","May","June","July","August","September","October","November","December"]
            stats = {m: {"students": set(), "total": 0.0} for m in months_order}
            
            for m, name, total in rows:
                if m in stats:
                    stats[m]["students"].add(name)
                    stats[m]["total"] += total
                    
            pdf = InvoicePDF()
            pdf.add_page()
            title = f"Monthly Report - Year {year}" if month_filter == "All" else f"Monthly Report - {month_filter} {year}"
            pdf.set_font("Times", 'B', 14)
            pdf.cell(0, 10, title, ln=True, align='C')
            pdf.ln(5)
            pdf.set_font("Times", 'B', 10)
            pdf.set_fill_color(26, 115, 232); pdf.set_text_color(255)
            pdf.cell(90, 10, " Month", 1, 0, 'L', True)
            pdf.cell(40, 10, "Students", 1, 0, 'C', True)
            pdf.cell(60, 10, "Total (MMK)", 1, 1, 'C', True)
            pdf.set_text_color(0); pdf.set_font("Times", '', 10)
            
            grand_students = 0
            grand_total = 0.0
            for m in months_order:
                if month_filter == "All" or m == month_filter:
                    s_count = len(stats[m]["students"])
                    t_sum = stats[m]["total"]
                    grand_students += s_count
                    grand_total += t_sum
                    pdf.cell(90, 9, f" {m}", 1)
                    pdf.cell(40, 9, f"{s_count}", 1, 0, 'C')
                    pdf.cell(60, 9, f"{t_sum:,.0f}", 1, 1, 'R')
            
            pdf.set_font("Times", 'B', 11)
            pdf.cell(90, 10, " TOTAL", 1)
            pdf.cell(40, 10, f"{grand_students}", 1, 0, 'C')
            pdf.cell(60, 10, f"{grand_total:,.0f}", 1, 1, 'R')
            
            out = f"Report_{year}.pdf" if month_filter == "All" else f"Report_{year}_{month_filter}.pdf"
            pdf.output(out)
            messagebox.showinfo("Success", f"Report generated: {out}")
            self.open_file(out)
        except Exception as e:
            messagebox.showerror("Report Error", f"Error: {e}")

            conn.close()

                elif len(parts) == 2:
                    items.append({"desc": parts[0], "amt": float(parts[1]), "adv": 0.0})
            
            # Use total and advance from DB
            total_db = float(rec[6])
            try:
                advance_db = float(rec[7]) if rec[7] is not None else 0.0
            except:
                advance_db = 0.0
            
            subtotal_db = total_db + advance_db
            
            old_items = self.current_items
            self.current_items = items
            try:
                self.export_invoice_pdf(rec[0], rec[2], rec[4], rec[5], subtotal_db, advance_db)
                messagebox.showinfo("Success", f"Re-printed Invoice #{rec[0]}")
            finally:
                self.current_items = old_items
        except Exception as e:
            messagebox.showerror("Reprint Error", f"Error: {e}")

    def delete_payment(self):
        sel = self.db_tree.selection()
        if not sel: return
        vals = self.db_tree.item(sel[0], "values")
        inv_id = vals[0]
        
        if not messagebox.askyesno("Confirm Delete", f"Are you sure you want to delete invoice #{inv_id}?"):
            return
            
        try:
            conn = sqlite3.connect(DB_FILE)
            cursor = conn.cursor()
            cursor.execute("DELETE FROM invoices WHERE id = ?", (inv_id,))
            conn.commit()
            conn.close()
            
            self.load_db_data()
            messagebox.showinfo("Success", f"Deleted invoice #{inv_id}")
        except Exception as e:
            messagebox.showerror("Delete Error", f"Error: {e}")

    def verify_invoice_ui(self):
        """UI for verifying an invoice's authenticity."""
        win = tk.Toplevel(self.root)
        win.title("Verify Invoice Authenticity")
        win.geometry("450x250")
        win.resizable(False, False)
        
        frm = tk.Frame(win, padx=20, pady=20)
        frm.pack(fill="both", expand=True)
        
        tk.Label(frm, text="Enter the Verification ID from the printed invoice:", font=("Arial", 10, "bold")).pack(pady=(0,10))
        tk.Label(frm, text="Format: 20260317XXXXXX-XXXXXXXXXXXX", fg="gray", font=("Arial", 8)).pack()
        
        ent_id = tk.Entry(frm, font=("Consolas", 12), justify="center", bd=2, relief="groove")
        ent_id.pack(fill="x", pady=15)
        ent_id.focus_set()
        
        def do_verify():
            raw = ent_id.get().strip()
            if "-" not in raw:
                messagebox.showwarning("Invalid Format", "Please enter the full Verification ID (including the dash).")
                return
            
            try:
                inv_id, provided_hash = raw.split("-", 1)
                expected_hash = calculate_verification_hash(inv_id)
                
                if provided_hash.upper() == expected_hash:
                    conn = sqlite3.connect(DB_FILE)
                    cursor = conn.cursor()
                    cursor.execute("SELECT COUNT(*) FROM invoices WHERE id = ?", (inv_id,))
                    exists = cursor.fetchone()[0] > 0
                    conn.close()
                    
                    if exists:
                        messagebox.showinfo("Authentic", f"✅ VERIFIED: This is an AUTHENTIC invoice issued by The SPARK System.\n\nInvoice ID: {inv_id}")
                    else:
                        messagebox.showwarning("Authentic but Not Found", f"⚠️ The Verification ID is mathematically correct, but Invoice #{inv_id} was NOT found in your current database.")
                else:
                    messagebox.showerror("FORGED", "❌ WARNING: This invoice appears to be FORGED or modified!")
            except Exception as e:
                messagebox.showerror("Error", f"An error occurred during verification: {e}")
        
        btn_verify = ttk.Button(frm, text="VERIFY NOW", style="Primary.TButton", command=do_verify)
        btn_verify.pack(pady=10)
        ent_id.bind("<Return>", lambda e: do_verify())

    def export_monthly_csv(self):
        year = str(self.ent_year.get())
        month = self.rep_month.get()
        out = f"Report_{year}.csv" if month == "All" else f"Report_{year}_{month}.csv"
        
        try:
            conn = sqlite3.connect(DB_FILE)
            cursor = conn.cursor()
            if month == "All":
                cursor.execute("SELECT * FROM invoices WHERE year = ?", (year,))
            else:
                cursor.execute("SELECT * FROM invoices WHERE year = ? AND month = ?", (year, month))
            rows = cursor.fetchall()
            conn.close()
            
            with open(out, 'w', newline='', encoding='utf-8') as f:
                w = csv.writer(f)
                w.writerow(["ID", "Date", "Name", "Items_Details", "Month", "Year", "Total", "Remarks"])
                for r in rows:
                    w.writerow(r)
            messagebox.showinfo("Success", f"CSV exported: {out}")
            self.open_file(out)
        except Exception as e:
            messagebox.showerror("CSV Error", f"Error: {e}")

    def backup_database(self):
        if not os.path.exists(DB_FILE):
            messagebox.showwarning("Backup Error", "Database not found.")
            return
        try:
            timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
            dest = filedialog.asksaveasfilename(defaultextension=".db", initialfile=f"Backup_{timestamp}.db")
            if dest:
                shutil.copy2(DB_FILE, dest)
                messagebox.showinfo("Backup Success", f"Database backed up to {dest}")
        except Exception as e:
            messagebox.showerror("Backup Error", f"Error: {e}")

    def restore_database(self):
        if not messagebox.askyesno("Confirm Restore", "Overwriting current database! Continue?"):
            return
        try:
            src = filedialog.askopenfilename(filetypes=[("SQLite DB", "*.db"), ("CSV Backup", "*.csv")])
            if src:
                if src.endswith(".db"):
                    shutil.copy2(src, DB_FILE)
                else:
                    # Restore from CSV (older backups)
                    conn = sqlite3.connect(DB_FILE)
                    cursor = conn.cursor()
                    cursor.execute("DELETE FROM invoices")
                    with open(src, 'r', encoding='utf-8') as f:
                        reader = csv.DictReader(f)
                        for row in reader:
                            cursor.execute("INSERT INTO invoices VALUES (?,?,?,?,?,?,?,?)", 
                                         (row["ID"], row["Date"], row["Name"], row["Items_Details"], row["Month"], row["Year"], float(row["Total"]), row.get("Remarks", "")))
                    conn.commit()
                    conn.close()
                self.load_db_data()
                messagebox.showinfo("Restore Success", "Database restored.")
        except Exception as e:
            messagebox.showerror("Restore Error", f"Error: {e}")

    def open_file(self, path):
        try:
            if os.name == "nt":
                os.startfile(path)
            else:
                import subprocess
                subprocess.Popen(["xdg-open", path])
        except:
            pass
    
    def update_status(self):
        try:
            count = len(self.db_tree.get_children())
            total = 0.0
            for iid in self.db_tree.get_children():
                vals = self.db_tree.item(iid, "values")
                if len(vals) >= 5:
                    try:
                        total += float(str(vals[4]).replace(",", ""))
                    except:
                        pass
            self.status_label.config(text=f"Records: {count} | Total: {total:,.0f} MMK")
        except:
            pass
        finally:
            try:
                self.root.after(1500, self.update_status)
            except:
                pass

    def load_credentials(self):
        default = {"user": "admin", "pass_hash": hashlib.sha256("Npsi@123".encode()).hexdigest()}
        if not os.path.exists(self.creds_file):
            self.save_credentials(default)
            return default
        try:
            with open(self.creds_file, 'r') as f:
                return json.load(f)
        except:
            return default

    def save_credentials(self, data):
        with open(self.creds_file, 'w') as f:

    def attempt_login(self, event=None):
        username = self.ent_user.get().strip()
        password = self.ent_pass.get().strip()
        pass_hash = hashlib.sha256(password.encode()).hexdigest()

        if username == self.credentials["user"] and pass_hash == self.credentials["pass_hash"]:
            # Clear login UI and start main app
            for widget in self.root.winfo_children():
                widget.destroy()
            EnterpriseApp(self.root)
        else:
            # Shake effect or better error message
            messagebox.showerror("Authentication Failed", "The credentials provided are incorrect.\nPlease try again.")
